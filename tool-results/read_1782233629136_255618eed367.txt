   150→  const [showSettings, setShowSettings] = useState(false);
   151→  const chatEndRef = useRef<HTMLDivElement>(null);
   152→  const inputRef = useRef<HTMLInputElement>(null);
   153→
   154→  useEffect(() => {
   155→    setMessages([
   156→      {
   157→        id: "welcome",
   158→        role: "assistant",
   159→        content:
   160→          "Bem-vindo ao **p3pp4** 🔮\n\nCole o link de qualquer simulador de física e eu resolvo automaticamente.\n\n• 🔬 Analiso o simulador com IA\n• 🧮 Calculo as respostas com precisão\n• 📸 Gero screenshot com seu nome\n• 📝 Explico passo a passo\n• 🤖 Múltiplos modelos de IA disponíveis",

   172→      const res = await fetch("/api/history");

   176→

   178→    const url = inputUrl.trim();
   179→    if (!url || (!url.startsWith("http://") && !url.startsWith("https://"))) {
   180→      toast.error("Cole uma URL que comece com http:// ou https://");
   181→      return;
   182→    }

   185→
   186→    const userMsg: ChatMessage = {
   187→      id: `user-${Date.now()}`,
   188→      role: "user",
   189→      content: url,
   190→      timestamp: new Date(),

   198→      "Analisando simulador...",
   199→      "Extraindo dados com Playwright...",
   200→      "Calculando respostas...",
   201→      "Preenchendo simulador...",
   202→      "Gerando screenshot...",
   203→      "Criando resumo didático...",

   206→    const stepInterval = setInterval(() => {
   207→      stepIdx = (stepIdx + 1) % steps.length;
   208→      setLoadingStep(steps[stepIdx]);
   209→    }, 4000);
   210→
   211→    try {
   212→      // Step 1: Start the job (returns immediately)
   213→      const startRes = await fetch("/api/solve/start", {
   214→        method: "POST",
   215→        headers: { "Content-Type": "application/json" },
   216→        body: JSON.stringify({ url, userName, model: selectedModel }),
   217→      });
   218→
   219→      if (!startRes.ok) {
   220→        throw new Error(`HTTP ${startRes.status}: ${startRes.statusText}`);
   221→      }
   222→
   223→      const startData = await startRes.json();
   224→      const jobId = startData.id;
   225→
   226→      if (!jobId) {
   227→        throw new Error("No job ID returned from server");
   228→      }
   229→
   230→      // Step 2: Poll for result every 3 seconds
   231→      const MAX_POLLS = 80; // 80 * 3s = 240s max
   232→      let pollCount = 0;
   233→
   234→      const pollForResult = async (): Promise<any> => {
   235→        pollCount++;
   236→        if (pollCount > MAX_POLLS) {
   237→          throw new Error("E_POLL_TIMEOUT");
   238→        }
   239→
   240→        await new Promise((r) => setTimeout(r, 3000));
   241→
   242→        const statusRes = await fetch(`/api/solve/status?id=${jobId}`);
   243→        if (!statusRes.ok) {
   244→          throw new Error(`Status check failed: HTTP ${statusRes.status}`);
   245→        }
   246→
   247→        const statusData = await statusRes.json();
   248→
   249→        if (statusData.complete) {
   250→          return statusData;
   251→        }
   252→
   253→        // Update loading step with progress
   254→        setLoadingStep(`Aguardando solver... (${pollCount * 3}s)`);
   255→
   256→        return pollForResult();
   257→      };
   258→
   259→      const data = await pollForResult();

   261→
   262→      const isCorrect = data.resultado?.includes("CORRETO");

   264→      const assistantMsg: ChatMessage = {
   265→        id: `ai-${Date.now()}`,
   266→        role: "assistant",
   267→        content: isCorrect
   268→          ? `✅ **Resolvido com sucesso!** O simulador foi resolvido corretamente.`
   269→          : `❌ **Resultado:** ${data.resultado || "Incorreto"}`,
   270→        result: data,
   271→        timestamp: new Date(),
   272→      };
   273→      setMessages((prev) => [...prev, assistantMsg]);
   274→      toast.success(isCorrect ? "Correto!" : "Verifique o resultado");
   275→    } catch (error: any) {
   276→      clearInterval(stepInterval);
   277→
   278→      const isTimeout = error.message?.includes("E_POLL_TIMEOUT");
   279→      const errorCode = isTimeout ? "E_POLL_TIMEOUT" : "E_FETCH_ERROR";
   280→
   281→      const assistantMsg: ChatMessage = {
   282→        id: `ai-${Date.now()}`,
   283→        role: "assistant",
   284→        content: `❌ **Erro ao resolver o simulador.**`,
   285→        result: {
   286→          success: false,
   287→          simulatorType: "",
   288→          enunciado: "",
   289→          variaveis: {},
   290→          calculos: "",
   291→          respostas: [],
   292→          respostasCorretas: {},
   293→          resultado: "ERRO",
   294→          metodo: "",
   295→          screenshotUrl: null,
   296→          errors: [errorCode],
   297→          technicalDetail: isTimeout
   298→            ? "Solver excedeu 240s de espera. O processo pode ter travado."
   299→            : error.message,
   300→          error: error.message,
   301→        },
   302→        timestamp: new Date(),
   303→      };
   304→      setMessages((prev) => [...prev, assistantMsg]);
   305→      toast.error("Erro — veja detalhes técnicos");
   306→    } finally {
   307→      setIsLoading(false);
   308→      setLoadingStep("");
   309→      inputRef.current?.focus();
   310→    }
   311→  };
   312→
   313→  const handleDeleteHistory = async (id: string) => {
   314→    try {
   315→      await fetch("/api/history", {
   316→        method: "DELETE",
   317→        headers: { "Content-Type": "application/json" },
   318→        body: JSON.stringify({ id }),
   319→      });
   320→      setHistory((prev) => prev.filter((h) => h.id !== id));
   321→    } catch {}
   322→  };
   323→
   324→  const renderErrorDetail = (result: SolverResult) => {
   325→    if (!result.errors || result.errors.length === 0) return null;
   326→
   327→    return (
   328→      <Card className="border-destructive/50 bg-destructive/5">
   329→        <CardHeader className="pb-2">
   330→          <CardTitle className="text-sm flex items-center gap-2 text-destructive">
   331→            <AlertTriangle className="w-4 h-4" />
   332→            Detalhes Técnicos do Erro
   333→          </CardTitle>
   334→        </CardHeader>
   335→        <CardContent className="space-y-3">
   336→          {result.errors.map((code) => {

   346→                    {info.title}
   347→                  </span>
   348→                </div>
   349→                <p className="text-xs text-muted-foreground ml-2">{info.desc}</p>
   350→                <p className="text-xs text-emerald-400 ml-2">
   351→                  💡 {info.fix}
   352→                </p>
   353→              </div>
   354→            );
   355→          })}
   356→          {result.technicalDetail && (
   357→            <div className="mt-2 p-2 rounded bg-black/30 border border-border">
   358→              <p className="text-[10px] font-mono text-muted-foreground break-all">
   359→                {result.technicalDetail}
   360→              </p>
   361→            </div>
   362→          )}
   363→        </CardContent>
   364→      </Card>

   366→  };
   367→
   368→  const renderResult = (result: SolverResult) => {

   395→              🤖 {result.model}
   396→            </Badge>
   397→          )}
   398→        </div>
   399→
   400→        {/* Screenshot */}
   401→        {result.screenshotUrl && (
   402→          <Card className="overflow-hidden border-purple-900/30 bg-card glow-purple">
   403→            <CardHeader className="pb-2">
   404→              <CardTitle className="text-sm flex items-center gap-2 text-purple-300">

   406→                Screenshot do Resultado
   407→              </CardTitle>
   408→            </CardHeader>

   410→              <img
   411→                src={result.screenshotUrl}

   415→            </CardContent>
   416→          </Card>
   417→        )}
   418→
   419→        {/* Respostas */}
   420→        {result.respostas.length > 0 && (
   421→          <Card className="border-purple-900/30 bg-card">
   422→            <CardHeader className="pb-2">
   423→              <CardTitle className="text-sm flex items-center gap-2 text-purple-300">
   424→                <Sparkles className="w-4 h-4" />
   425→                Respostas
   426→              </CardTitle>
   427→            </CardHeader>
   428→            <CardContent>
   429→              <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
   430→                {result.respostas.map((resp, i) => {
   431→                  const correctKey = `Answer${i + 1}`;
   432→                  const correctVal = result.respostasCorretas?.[correctKey];
   433→                  return (
   434→                    <div
   435→                      key={i}
   436→                      className="flex items-center justify-between p-2.5 rounded-lg bg-muted/50 border border-border"

   438→                      <span className="text-sm text-muted-foreground">Campo {i + 1}</span>
   439→                      <div className="text-right">
   440→                        <span className="font-mono font-bold text-foreground">{resp}</span>

   442→                          <span className="text-[11px] text-muted-foreground ml-2">
   443→                            (exato: {typeof correctVal === "number" ? correctVal.toPrecision(4) : correctVal})
   444→                          </span>
   445→                        )}
   446→                      </div>
   447→                    </div>
   448→                  );
   449→                })}
   450→              </div>
   451→            </CardContent>
   452→          </Card>
   453→        )}
   454→
   455→        {/* Cálculos / Resumo */}
   456→        {result.calculos && (
   457→          <Card className="border-purple-900/30 bg-card">
   458→            <CardHeader className="pb-2">
   459→              <CardTitle className="text-sm flex items-center gap-2 text-purple-300">
   460→                📝 Resumo da Resolução
   461→              </CardTitle>
   462→            </CardHeader>
   463→            <CardContent>
   464→              <div className="prose prose-sm prose-invert max-w-none">
   465→                {result.calculos.split("\n").map((line, i) => {

   467→                    return <h3 key={i} className="text-purple-300 font-bold mt-4 mb-1 text-sm">{line.replace("## ", "")}</h3>;
   468→                  if (line.startsWith("### "))
   469→                    return <h4 key={i} className="text-violet-300 font-semibold mt-3 mb-1 text-sm">{line.replace("### ", "")}</h4>;

   471→                    return <strong key={i} className="text-purple-200">{line.replace(/\*\*/g, "")}</strong>;

   475→                    return null;
   476→                  if (line.match(/^\d+\./))
   477→                    return <li key={i} className="ml-4 text-sm list-decimal text-muted-foreground">{line.replace(/^\d+\.\s*/, "")}</li>;
   478→                  return line ? <p key={i} className="text-sm leading-relaxed text-muted-foreground">{line}</p> : null;
   479→                })}
   480→              </div>
   481→            </CardContent>
   482→          </Card>
   483→        )}
   484→
   485→        {/* Error details */}
   486→        {renderErrorDetail(result)}
   487→      </div>
   488→    );
   489→  };
   490→
   491→  return (
   492→    <div className="min-h-screen flex flex-col bg-background">
   493→      {/* Header */}
   494→      <header className="sticky top-0 z-50 border-b border-purple-900/20 bg-background/90 backdrop-blur-md">
   495→        <div className="max-w-4xl mx-auto px-4 py-3 flex items-center justify-between">
   496→          <div className="flex items-center gap-3">
   497→            <div className="w-9 h-9 rounded-xl bg-gradient-to-br from-purple-600 to-fuchsia-600 flex items-center justify-center shadow-lg shadow-purple-900/30">
   498→              <Skull className="w-5 h-5 text-white" />
   499→            </div>
   500→            <div>
   501→              <h1 className="font-bold text-lg leading-tight glow-text text-purple-100">
   502→                p3pp4
   503→              </h1>
   504→              <p className="text-[10px] text-purple-400/60">
   505→                AI Physics Solver
   506→              </p>
   507→            </div>
   508→          </div>
   509→
   510→          <div className="flex items-center gap-2">
   511→            {/* Model selector */}
   512→            <Select value={selectedModel} onValueChange={setSelectedModel}>

   519→                    <span className="font-semibold">{m.name}</span>
   520→                  </SelectItem>
   521→                ))}
   522→              </SelectContent>
   523→            </Select>
   524→
   525→            {/* Settings */}
   526→            <Dialog open={showSettings} onOpenChange={setShowSettings}>
   527→              <DialogTrigger asChild>
   528→                <Button variant="outline" size="sm" className="gap-1.5 border-purple-900/30 text-purple-300 hover:bg-purple-900/20 h-8">
   529→                  <Settings className="w-3.5 h-3.5" />
   530→                  <span className="hidden sm:inline text-xs">{userName}</span>
   531→                </Button>
   532→              </DialogTrigger>
   533→              <DialogContent className="bg-card border-purple-900/30">
   534→                <DialogHeader>
